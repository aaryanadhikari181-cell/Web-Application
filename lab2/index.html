Exploring core JavaScript concepts: var/let/const, functions, arrow functions, objects, map, filter, and spread operators.

01

var, let & const — Variable Declarations

📝 Code
// var — function-scoped, can be re-declared
var city = "Kathmandu";
var city = "Pokhara"; // re-declaration allowed
console.log(city);

// let — block-scoped, can be reassigned
let age = 20;
age = 21; // reassignment OK
console.log(age);

// const — block-scoped, cannot be reassigned
const PI = 3.14159;
console.log(PI);

// Block scope demo
{
  let blockVar = "inside block";
  console.log(blockVar);
}
// console.log(blockVar); // Error! Not accessible here
💻 Output
// city (var re-declared)
"Pokhara"
// age (let reassigned)
21
// PI (const)
3.14159
// block-scoped let
"inside block"
02

Regular Functions

📝 Code
// Function declaration
function greet(name) {
  return `Hello, ${name}!`;
}
console.log(greet("Aaryan"));

// Function with default parameter
function add(a, b = 10) {
  return a + b;
}
console.log(add(5));       // 15 (b defaults to 10)
console.log(add(5, 3));    // 8

// Function expression
const multiply = function(x, y) {
  return x * y;
};
console.log(multiply(4, 5));
💻 Output
"Hello, Aaryan!"
15
8
20

Arrow Functions

📝 Code
// Arrow function — concise syntax
const square = (n) => n * n;
console.log(square(6));  // 36

// Multi-param arrow function
const fullName = (first, last) => `${first} ${last}`;
console.log(fullName("Aaryan", "Adhikari"));

// Arrow with block body
const isAdult = (age) => {
  if (age >= 18) return "Adult";
  return "Minor";
};
console.log(isAdult(20));

// No params
const sayHi = () => "Hi there!";
console.log(sayHi());

// Difference: arrow functions don't have own 'this'
const obj = {
  name: "Aaryan",
  regular: function() { return this.name; },
  arrow:   () => "arrow has no own 'this'"
};
console.log(obj.regular());
console.log(obj.arrow());
💻 Output
36
"Aaryan Adhikari"
"Adult"
"Hi there!"
"Aaryan"
"arrow has no own 'this'"

Objects

📝 Code
// Object literal
const student = {
  name: "Aaryan",
  age: 20,
  city: "Kathmandu",
  introduce() {
    return `I'm ${this.name}, ${this.age} years old.`;
  }
};

// Access properties
console.log(student.name);         // dot notation
console.log(student["city"]);     // bracket notation
console.log(student.introduce());

// Destructuring
const { name, age } = student;
console.log(name, age);

// Add / update properties
student.email = "aaryan@example.com";
student.age = 21;
console.log(student.email, student.age);

// Object.keys / Object.values
console.log(Object.keys(student));
console.log(Object.values(student));
💻 Output
"Aaryan"
"Kathmandu"
"I'm Aaryan, 20 years old."
"Aaryan" 20
"aaryan@example.com" 21

Array.map() — Transform Arrays

📝 Code
const numbers = [1, 2, 3, 4, 5];

// Double each number
const doubled = numbers.map(n => n * 2);
console.log(doubled);

// Square each number
const squared = numbers.map(n => n ** 2);
console.log(squared);

// Map over objects
const students = [
  { name: "Aaryan", score: 85 },
  { name: "Priya",  score: 92 },
  { name: "Rajan",  score: 78 }
];

const names = students.map(s => s.name);
console.log(names);

const withGrade = students.map(s => ({
  ...s,
  grade: s.score >= 90 ? "A" : s.score >= 80 ? "B" : "C"
}));
console.log(withGrade);
💻 Output
[2, 4, 6, 8, 10]
[1, 4, 9, 16, 25]
["Aaryan", "Priya", "Rajan"]
[
  {name:"Aaryan", score:85, grade:"B"},
  {name:"Priya", score:92, grade:"A"},
  {name:"Rajan", score:78, grade:"C"}
]

Array.filter() — Select Items

📝 Code
const numbers = [1, 2, 3, 4, 5, 6, 7, 8];

// Filter even numbers
const evens = numbers.filter(n => n % 2 === 0);
console.log(evens);

// Filter numbers greater than 4
const big = numbers.filter(n => n > 4);
console.log(big);

// Filter objects
const products = [
  { name: "Laptop", price: 1200, inStock: true },
  { name: "Mouse",  price: 25,   inStock: false },
  { name: "Monitor",price: 350,  inStock: true },
  { name: "Keyboard",price: 80,  inStock: true }
];

// Only in-stock items
const available = products.filter(p => p.inStock);
console.log(available.map(p => p.name));

// Affordable (under 400) AND in stock
const deals = products.filter(p => p.price < 400 && p.inStock);
console.log(deals.map(p => p.name));
💻 Output
[2, 4, 6, 8]
[5, 6, 7, 8]
["Laptop", "Monitor", "Keyboard"]
["Monitor", "Keyboard"]

Spread Operator (...)

📝 Code
// Spread in arrays — combine arrays
const a = [1, 2, 3];
const b = [4, 5, 6];
const combined = [...a, ...b];
console.log(combined);

// Clone an array (no reference)
const original = [10, 20, 30];
const clone = [...original];
clone.push(40);
console.log(original); // unchanged
console.log(clone);

// Spread in objects — merge objects
const defaults = { theme: "dark", lang: "en" };
const userPrefs = { lang: "np", fontSize: 16 };
const settings = { ...defaults, ...userPrefs };
console.log(settings);

// Clone & update object
const user = { name: "Aaryan", age: 20 };
const updated = { ...user, age: 21, city: "Pokhara" };
console.log(updated);

// Spread in function calls
const nums = [3, 1, 4, 1, 5, 9];
console.log(Math.max(...nums));
💻 Output
[1, 2, 3, 4, 5, 6]
[10, 20, 30]
[10, 20, 30, 40]
{theme:"dark", lang:"np", fontSize:16}
{name:"Aaryan", age:21, city:"Pokhara"}
